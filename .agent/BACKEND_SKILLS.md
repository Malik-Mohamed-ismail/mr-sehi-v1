## Core Rules (apply to every table)

- All monetary values: `decimal(12, 4)` — 4 decimal places, no floating point errors
- All timestamps: `timestamp with time zone` (TIMESTAMPTZ) — stored UTC, displayed in Asia/Riyadh
- Every financial table has: `created_by INTEGER REFERENCES users(id)` + `updated_at TIMESTAMPTZ`
- Soft deletes only where indicated — never hard-delete financial records
- Foreign key constraints enforced at DB level, not just application layer

## 🔧 FIX: `purchase_invoices` — No GENERATED Columns

Original spec used `GENERATED ALWAYS AS` for `subtotal` and `total_amount`. This is broken because `vat_amount` is a user input, creating a circular dependency. **Fix: compute all derived values in the application layer** before inserting.

```typescript
// Application-layer computation (in purchases.service.ts):
const subtotal     = quantity * unit_price - discount
const vat          = calculateVAT(subtotal, supplierHasVAT(supplier.vat_number))
const total_amount = vat.total
// Insert these pre-computed values — no generated columns
```

## 🔧 FIX: `journal_entries.is_balanced` — No GENERATED Column

PostgreSQL cannot use correlated subqueries in `GENERATED ALWAYS AS` expressions. **Fix: service sets `is_balanced = true` only after it validates that sum(debit) === sum(credit) across all lines.**

## Auth Middleware

```typescript
// /src/middleware/auth.ts
export const authenticate = (req: Request, res: Response, next: NextFunction) => {
  const token = req.headers.authorization?.split(' ')[1]
  if (!token) throw new AppError('UNAUTHORIZED', 401, 'مطلوب تسجيل الدخول')
  const payload = jwt.verify(token, env.JWT_SECRET) as JWTPayload
  req.user = { id: payload.sub, role: payload.role, username: payload.username }
  next()
}

// /src/middleware/authorize.ts
export const authorize = (...roles: UserRole[]) =>
  (req: Request, res: Response, next: NextFunction) => {
    if (!roles.includes(req.user.role))
      throw new AppError('FORBIDDEN', 403, 'ليس لديك صلاحية للوصول')
    next()
  }
```

```typescript
export const VAT_RATE = 0.15   // ZATCA Saudi VAT — 15%

export function supplierHasVAT(vatNumber: string | null): boolean {
  // العزيزية (vegetable market) has no VAT number → no VAT applied
  return !!vatNumber && vatNumber !== '0' && vatNumber.trim() !== ''
}

export function calculateVAT(subtotal: number, hasVAT: boolean) {
  if (!hasVAT) return { subtotal, vatAmount: 0, total: subtotal, hasVAT: false }
  const vatAmount = parseFloat((subtotal * VAT_RATE).toFixed(4))
  const total     = parseFloat((subtotal + vatAmount).toFixed(4))
  return { subtotal, vatAmount, total, hasVAT: true }
}
```

```typescript
// /src/modules/purchases/purchases.service.ts
export async function createPurchaseInvoice(dto: CreatePurchaseDto, userId: number) {
  return await db.transaction(async (tx) => {
    // 1. Resolve supplier VAT status
    const [supplier] = await tx.select().from(suppliers)
      .where(eq(suppliers.id, dto.supplier_id))
    const vat = calculateVAT(dto.subtotal, supplierHasVAT(supplier?.vat_number))

    // 2. Validate submitted amounts match server-computed amounts
    if (Math.abs(vat.vatAmount - dto.vat_amount) > 0.01)
      throw new AppError('VALIDATION_ERROR', 400, 'مبلغ الضريبة لا يتطابق مع الحساب')

    // 3. Insert invoice with pre-computed amounts
    const [invoice] = await tx.insert(purchaseInvoices).values({
      ...dto,
      subtotal:     vat.subtotal,
      vat_amount:   vat.vatAmount,
      total_amount: vat.total,
      created_by:   userId
    }).returning()

    // 4. Auto-generate balanced journal entry
    const entry = await createPurchaseJournalEntry(tx, invoice, supplier)

    // 5. Link entry back to invoice
    await tx.update(purchaseInvoices)
      .set({ journal_entry_id: entry.id })
      .where(eq(purchaseInvoices.id, invoice.id))

    // 6. Audit trail
    await writeAuditLog(tx, {
      userId, action: 'CREATE',
      tableName: 'purchase_invoices',
      recordId: invoice.id,
      newValues: invoice
    })

    return { ...invoice, journal_entry_id: entry.id }
  })
}
```

```typescript
// /src/modules/purchases/purchases.accounting.ts

async function createPurchaseJournalEntry(tx, invoice, supplier) {
  const lines = []

  // DR: Inventory or Asset account based on category
  const debitAccount = invoice.is_asset
    ? getAssetAccount(invoice.category)
    : getInventoryAccount(invoice.category)
  lines.push({ account_code: debitAccount, debit_amount: invoice.subtotal, credit_amount: 0 })

  // DR: VAT Input (account 1110) — if applicable
  if (Number(invoice.vat_amount) > 0)
    lines.push({ account_code: '1110', debit_amount: Number(invoice.vat_amount), credit_amount: 0 })

  // CR: Cash / Bank / Suppliers Payable — based on payment method
  const creditAccount = {
    'كاش': '1101',    // Petty cash
    'بنك': '1104',    // Bank
    'آجل': '2101',    // Accounts payable
  }[invoice.payment_method]
  lines.push({ account_code: creditAccount, debit_amount: 0, credit_amount: Number(invoice.total_amount) })

  // Validate balance (hard error if off by > 0.001)
  const totalDebit  = lines.reduce((s, l) => s + Number(l.debit_amount),  0)
  const totalCredit = lines.reduce((s, l) => s + Number(l.credit_amount), 0)
  if (Math.abs(totalDebit - totalCredit) > 0.001)
    throw new AppError('JOURNAL_UNBALANCED', 422, 'القيد غير متوازن')

  const entryNumber = await generateEntryNumber(tx, 'P')
  const [entry] = await tx.insert(journalEntries).values({
    entry_number: entryNumber,
    entry_date:   invoice.invoice_date,
    description:  `شراء ${invoice.item_name} — ${supplier.name_ar}`,
    reference:    invoice.invoice_number,
    source_type:  'purchase',
    source_id:    invoice.id,
    is_balanced:  true,
  }).returning()

  await tx.insert(journalEntryLines).values(
    lines.map(l => ({ ...l, entry_id: entry.id }))
  )
  return entry
}
```

## Journal Entry Reversal

```typescript
// /src/modules/journal/journal.service.ts
export async function reverseJournalEntry(entryId: number, userId: number, reason: string) {
  return await db.transaction(async (tx) => {
    const [original] = await tx.select().from(journalEntries)
      .where(eq(journalEntries.id, entryId))
    if (!original)            throw new AppError('NOT_FOUND',    404, 'القيد غير موجود')
    if (original.is_reversed) throw new AppError('ENTRY_LOCKED', 423, 'القيد مُعكوس بالفعل')

    const originalLines = await tx.select().from(journalEntryLines)
      .where(eq(journalEntryLines.entry_id, entryId))
    const reversalNumber = await generateEntryNumber(tx, 'REV')

    const [reversal] = await tx.insert(journalEntries).values({
      entry_number: reversalNumber,
      entry_date:   new Date().toISOString().split('T')[0],
      description:  `عكس قيد: ${original.entry_number} — ${reason}`,
      reference:    original.entry_number,
      source_type:  'reversal',
      source_id:    original.id,
      is_balanced:  true,
      created_by:   userId,
    }).returning()

    // Swap debit ↔ credit on every line
    await tx.insert(journalEntryLines).values(
      originalLines.map(l => ({
        entry_id:      reversal.id,
        account_code:  l.account_code,
        debit_amount:  l.credit_amount,   // swapped
        credit_amount: l.debit_amount,    // swapped
        description:   `عكس: ${l.description ?? ''}`,
      }))
    )

    // Mark original as reversed
    await tx.update(journalEntries)
      .set({ is_reversed: true, reversed_by: reversal.id })
      .where(eq(journalEntries.id, entryId))

    return reversal
  })
}
```

## Entry Number Generator

```typescript
// /src/utils/entryNumberGenerator.ts
// Format: {PREFIX}-{YEAR}-{SEQ:04d}
// Examples: P-2026-0042 / R-2026-0015 / E-2026-0007 / REV-2026-0003

export async function generateEntryNumber(
  tx: DrizzleTransaction,
  prefix: 'P' | 'R' | 'E' | 'M' | 'REV'
) {
  const year    = new Date().getFullYear()
  const pattern = `${prefix}-${year}-%`

  const [last] = await tx
    .select({ n: journalEntries.entry_number })
    .from(journalEntries)
    .where(like(journalEntries.entry_number, pattern))
    .orderBy(desc(journalEntries.entry_number))
    .limit(1)

  const seq = last ? parseInt(last.n.split('-')[2]) + 1 : 1
  return `${prefix}-${year}-${String(seq).padStart(4, '0')}`
}
```

|--------|--------|

| `REV` | Reversal entry |

## Break-Even Analysis

```typescript
// /src/utils/accounting.ts
export function calculateBreakEven(
  totalRevenue: number,
  totalCOGS:    number,
  fixedCosts:   number
) {
  const grossProfit     = totalRevenue - totalCOGS
  const grossMarginPct  = totalRevenue > 0 ? grossProfit / totalRevenue : 0
  const breakEvenSales  = grossMarginPct > 0 ? fixedCosts / grossMarginPct : Infinity
  const safetyMarginSAR = totalRevenue - breakEvenSales
  const safetyMarginPct = totalRevenue > 0 ? safetyMarginSAR / totalRevenue : 0

  return {
    fixedCosts:         parseFloat(fixedCosts.toFixed(4)),
    grossMarginPct:     parseFloat(grossMarginPct.toFixed(4)),
    breakEvenSales:     parseFloat(breakEvenSales.toFixed(4)),
    currentRevenue:     totalRevenue,
    safetyMarginSAR:    parseFloat(safetyMarginSAR.toFixed(4)),
    safetyMarginPct:    parseFloat(safetyMarginPct.toFixed(4)),
    isBeyondBreakEven:  totalRevenue > breakEvenSales,
  }
}
```

```typescript
export const errorHandler: ErrorRequestHandler = (err, req, res, _next) => {
  const requestId = req.headers['x-request-id'] ?? crypto.randomUUID()

  // PostgreSQL unique violation
  if (err.code === '23505')
    return res.status(409).json({
      success: false,
      error: { code: 'DUPLICATE_ENTRY', message: 'هذا السجل موجود مسبقاً' },
      requestId,
    })

  // Zod validation error
  if (err.name === 'ZodError')
    return res.status(400).json({
      success: false,
      error: {
        code: 'VALIDATION_ERROR',
        message: 'بيانات غير صحيحة',
        details: err.errors.map((e: ZodIssue) => ({
          field: e.path.join('.'),
          message: e.message,
        })),
      },
      requestId,
    })

  // Application error (known)
  if (err instanceof AppError) {
    logger.warn({ err, requestId })
    return res.status(err.statusCode).json({
      success: false,
      error: { code: err.code, message: err.messageAr, details: err.details },
      requestId,
    })
  }

  // Unknown / unexpected
  logger.error({ err, requestId }, 'Unhandled error')
  return res.status(500).json({
    success: false,
    error: { code: 'INTERNAL_ERROR', message: 'حدث خطأ غير متوقع' },
    requestId,
  })
}
```

# B15. KEY ARCHITECTURE DECISIONS
# ═══════════════════════════════════════════════

| Decision | Choice | Rationale |
|----------|--------|-----------|
| Excel Export | **Frontend (SheetJS)** | Data in cache, instant download, RTL worksheets |
| PDF Export | **Backend (Puppeteer)** | Arabic font embedding, multi-page layout |
| ORM | **Drizzle ORM** | Type-safe, schema-as-code, no magic, no injection |
| Logger | **Pino** | Fastest Node.js logger, structured JSON for prod |
| Test Framework | **Vitest** | ESM-native, faster than Jest |
| Token Delivery | **HttpOnly Cookie** | Prevents XSS token theft |
| Number Display | **English digits always** | Saudi business standard (en-US locale) |
| Font (PDF) | **IBM Plex Sans Arabic** | Best bilingual AR/EN coverage, embedded in PDF |
| Cron Timezone | **Asia/Riyadh** | Subscriber status updated at midnight Saudi time |
| Amount Precision | **decimal(12, 4)** | No floating point errors on financial data |
| Generated Columns | **Removed → app layer** | PostgreSQL GENERATED cannot use correlated subqueries |
| Audit Trail | **Required on all CRUDs** | Financial compliance — every change tracked |
